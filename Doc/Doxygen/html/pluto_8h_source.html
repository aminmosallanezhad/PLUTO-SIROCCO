659/* ********************************************************
660 Define mass fractions (H_MASS_FRAC = xH and
661 He_MASS_FRAC = xHe):
662
663 xH = mH / mtot
664 xHe = mHe / mtot
665 xZ = mZ / mtot
666
667 where xH + xHe + xZ = 1.
668 Number fraction are defined from mass fraction in
669 pluto.h
670
671 - For SNEq, mass fractions are chosen so as to preserve
672 the original Raymond hydrogen number fraction (fZ = 1.e-3,
673 fHe = 0.082):
674
675 xH = 1 - xZ - xHe = 1 - xH/AH*(fZ*AZ + fHe*AHe)
676
677 giving xH = AH/(AH + fZ*AZ + fHe*AHe) while
678 xHe = fHe*AHe*xH/AH.
679
680 - For H2_COOL, Proto-Solar Mass Fractions for Hydrogen
681 and Helium (Lodders, ApJ 591, 2003 ) are used.
682
683 ******************************************************** */
684
685#ifndef H_MASS_FRAC /* Set default values */
686 #if COOLING == SNEq
687 #define H_MASS_FRAC 0.737743792120194
688 #else
689 #define H_MASS_FRAC 0.7110
690 #endif
691#endif
692
693#ifndef He_MASS_FRAC
694 #if (EOS == PVTE_LAW) && (COOLING == NO)
695 #define He_MASS_FRAC (1 - H_MASS_FRAC) /* Effective Y and not 0.2741
696 Baraffe (2008) */
697 #elif COOLING == SNEq
698 #define He_MASS_FRAC (0.082*CONST_AHe*H_MASS_FRAC/CONST_AH)
699 #else
700 #define He_MASS_FRAC 0.2741
701 #endif
702#endif
703
704/* *********************************************************************
705 Define number fractions (FRAC_He = fHe and FRAC_Z = fZ, with
706 respect to hydrogen - FRAC_H = 1):
707
708 fH = NH/NH = 1, fHe = NHe/NH, fZ = NZ/NH
709 ********************************************************************* */
710
711#define Z_MASS_FRAC (1.0 - H_MASS_FRAC - He_MASS_FRAC)
712#define FRAC_He (He_MASS_FRAC/CONST_AHe*CONST_AH/H_MASS_FRAC)
713#define FRAC_Z (Z_MASS_FRAC /CONST_AZ *CONST_AH/H_MASS_FRAC)
